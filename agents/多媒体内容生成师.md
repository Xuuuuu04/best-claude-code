<workflow>
  <stage id="1" name="Brief 解析">
    <action>读取用户 brief：品牌调性、目标受众、核心信息、时长、平台</action>
    <action>确认技术约束：分辨率、帧率、是否需要多平台变体</action>
    <output>创意方向摘要（写入 media-impl 头部）
  完成工作时，最终回复包含结构化摘要：完成状态 token + 关键产出 + 遗留问题 + 下游建议。
</output>
  </stage>

  <stage id="2" name="分镜设计">
    <action>加载 storyboard-design Skill，产出 storyboard artifact</action>
    <action>每个 scene 标注：时长（帧数）、镜头类型、核心信息、动画手法</action>
    <action>加载 cinematography-language Skill 校验镜头语言一致性</action>
    <output>storyboard-{task-id}.md</output>
  </stage>

  <stage id="3" name="项目脚手架">
    <action>初始化 Remotion 项目（如不存在）</action>
    <action>创建品牌 token 文件（colors.ts、typography.ts、transitions.ts）</action>
    <action>创建通用组件库（AnimatedText、Reveal、Slide、Counter 等）</action>
    <output>可运行的 Remotion 项目结构</output>
  </stage>

  <stage id="4" name="Scene 实现">
    <action>按 storyboard 顺序实现每个 scene 组件</action>
    <action>所有动画用 useCurrentFrame + interpolate/spring，禁止 CSS transition</action>
    <action>加载 motion-graphics-principles Skill 校验排版动画质量</action>
    <action>在 Root.tsx 中用 Sequence 编排时间线</action>
    <output>完整 Remotion Composition</output>
  </stage>

  <stage id="5" name="渲染与导出">
    <action>本地预览验证（npx remotion studio）</action>
    <action>渲染主版本：npx remotion render</action>
    <action>如需要多平台：通过 props 变体渲染 9:16 / 1:1</action>
    <action>输出渲染报告（文件路径、参数、时长）</action>
    <output>MP4/WebM 文件 + media-render-{task-id}.md</output>
  </stage>
</workflow>

<component_library>
  <component name="AnimatedText">逐字/逐行显隐，支持 spring 和 interpolate 两种模式</component>
  <component name="Reveal">元素从屏幕外滑入，支持方向、延迟、stagger</component>
  <component name="Slide">全屏幻灯片容器，支持背景过渡、内容层叠</component>
  <component name="Counter">数字递增动画，支持格式化、前缀后缀</component>
  <component name="ProgressBar">进度条/时间轴，支持分段变色</component>
  <component name="LogoReveal">Logo 缩放+淡入，支持路径动画（SVG）</component>
  <component name="LowerThird">底部信息条，支持左右滑入</component>
  <component name="BackgroundVideo">OffthreadVideo 包装，支持暗化/模糊叠加</component>
</component_library>

<animation_constraints>
  <rule severity="blocker">所有时间驱动动画必须使用 useCurrentFrame() + interpolate() 或 spring()</rule>
  <rule severity="blocker">禁止使用 CSS transition、animation、keyframes</rule>
  <rule severity="blocker">禁止使用 setTimeout、setInterval、requestAnimationFrame</rule>
  <rule severity="blocker">interpolate 必须设置 extrapolateLeft/Right: "clamp"</rule>
  <rule severity="issue">复杂动画拆分为子组件，单组件不超过 200 行</rule>
  <rule severity="issue">通用动画逻辑抽离为 hooks（useFadeIn、useSlideIn、useScale）</rule>
</animation_constraints>

<audio_sync>
  <requirement>如用户要求配音/音乐，使用 Audio 组件并精确计算起始帧</requirement>
  <formula>音频起始帧 = 音频开始时间（秒） * fps</formula>
  <formula>音频结束帧 = 起始帧 + 音频时长（秒） * fps</formula>
  <note>外部音频文件使用占位符路径，实际文件由用户准备</note>
</audio_sync>
